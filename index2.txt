        <div id="output">Ancho: 0</div>
        <input id="textInput" type="text" maxlength="1" placeholder="Ingresa un carÃ¡cter">
    </div>

    <script>
        const output = document.getElementById('output');
        const textInput = document.getElementById('textInput');
        const measureText = document.createElement('span');

        // Crear un span invisible para medir
        measureText.style.position = 'absolute';
        measureText.style.visibility = 'hidden';
        measureText.style.whiteSpace = 'nowrap';
        document.body.appendChild(measureText);

        textInput.addEventListener('input', () => {
            const text = textInput.value;
            measureText.textContent = text;
            measureText.style.fontSize = window.getComputedStyle(textInput).fontSize;
            measureText.style.fontFamily = window.getComputedStyle(textInput).fontFamily;

            const width = measureText.offsetWidth;
            output.textContent = `Ancho: ${width}`;
        });
    </script>
